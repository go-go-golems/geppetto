   212	func (m *moduleRuntime) attachRef(o *goja.Object, ref any) {
   213		// Set the value first so goja stores the Go pointer as-is (m.vm.ToValue
   214		// would wrap struct pointers in a proxy whose Export() returns a map).
   215		// Then redefine the property to make it non-enumerable/non-writable/non-configurable.
   216		_ = o.Set(hiddenRefKey, ref)
   217		_ = o.DefineDataProperty(hiddenRefKey, o.Get(hiddenRefKey),
   218			goja.FLAG_FALSE, // writable

   103		}
   104		handlerValue := obj.Get("handler")

   107			return fmt.Errorf("tool %s handler must be a function", name)

   110		fn := func(goCtx context.Context, in map[string]any) (any, error) {

   116				toolCtx["sessionId"] = sessionID

   119				toolCtx["inferenceId"] = inferenceID

   122				toolCtx["tags"] = cloneJSONMap(tags)

   125				if call.ID != "" {
   126					toolCtx["callId"] = call.ID
   127				}
   128				if call.Name != "" {
   129					toolCtx["callName"] = call.Name
   130				}
   131			}
   132			if deadline, ok := goCtx.Deadline(); ok {
   133				toolCtx["deadlineMs"] = deadline.UnixMilli()
   134			}
   135			retAny, err := r.api.callOnOwner(goCtx, "tools.register.handler", func(context.Context) (any, error) {

   146				return nil, fmt.Errorf("js tool %s: %w", name, err)
   147			}

   149		}

   151		if err != nil {
   152			return err
   153		}

   169			return err
   170		}

   222				"name":   e.ToolName,
   223				"status": e.Status,

   110	func (w *JSEmbeddingsWrapper) generateEmbeddingWithCallbacks(call goja.FunctionCall) goja.Value {
   111		if len(call.Arguments) < 2 {
   112			panic(w.runtime.NewTypeError("generateEmbeddingWithCallbacks requires text and callbacks arguments"))
   113		}
   114	
   115		text := call.Arguments[0].String()
   116		callbacksObj := call.Arguments[1].ToObject(w.runtime)
   117	
   118		var onSuccess, onError goja.Callable
   119		var ok bool
   120	
   121		if onSuccessVal := callbacksObj.Get("onSuccess"); onSuccessVal != nil {
   122			if onSuccess, ok = goja.AssertFunction(onSuccessVal); !ok {
   123				panic(w.runtime.NewTypeError("onSuccess must be a function"))
   124			}
   125		}
   126	
   127		if onErrorVal := callbacksObj.Get("onError"); onErrorVal != nil {
   128			if onError, ok = goja.AssertFunction(onErrorVal); !ok {
   129				panic(w.runtime.NewTypeError("onError must be a function"))
   130			}
   131		}
   132	
   133		var (
   134			cancelMu sync.Mutex
   135			cancelFn context.CancelFunc
   136		)
   137		setCancel := func(fn context.CancelFunc) {
   138			cancelMu.Lock()
   139			cancelFn = fn
   140			cancelMu.Unlock()
   141		}
   142		invokeCancel := func() {
   143			cancelMu.Lock()
   144			fn := cancelFn
   145			cancelMu.Unlock()
   146			if fn != nil {
   147				fn()
   148			}
   149		}
   150	
   151		w.loop.RunOnLoop(func(*goja.Runtime) {
   152			go func() {
   153				ctx, cancel := context.WithCancel(context.Background())
   154				setCancel(cancel)
   155				defer func() {
   156					cancel()
   157					setCancel(nil)
   158				}()
   159	
   160				embedding, err := w.provider.GenerateEmbedding(ctx, text)
   161				if err != nil {
   162					if onError != nil {
   163						w.loop.RunOnLoop(func(*goja.Runtime) {
   164							_, callErr := onError(goja.Undefined(), w.runtime.ToValue(err.Error()))
   165							if callErr != nil {
   166								log.Error().Err(callErr).Msg("failed to call onError")
   167							}
   168						})
   169					}
   170					return
   171				}
   172	
   173				if onSuccess != nil {
   174					// Convert []float32 to []interface{} for Goja
   175					embeddingInterface := make([]interface{}, len(embedding))
   176					for i, v := range embedding {
   177						embeddingInterface[i] = v
   178					}
   179	
   180					w.loop.RunOnLoop(func(*goja.Runtime) {
   181						_, callErr := onSuccess(goja.Undefined(), w.runtime.ToValue(embeddingInterface))
   182						if callErr != nil {
   183							log.Error().Err(callErr).Msg("failed to call onSuccess")
   184						}
   185					})
   186				}
   187			}()
   188		})
   189	
   190		// Return cancel function
   191		return w.runtime.ToValue(func(call goja.FunctionCall) goja.Value {
   192			invokeCancel()
   193			return goja.Undefined()
   194		})
   195	}

    55		if f == nil || f.providers == nil || f.spec == nil {
    56			return nil, fmt.Errorf("xgoja runtime factory is not initialized")
    57		}

    60			return nil, fmt.Errorf("unknown runtime profile %q", profile)

    72			engine.WithDataOnlyDefaultRegistryModules(false),

    76		}
    77		runtimeFactory, err := builder.Build()
    78		if err != nil {
    79			return nil, err
    80		}

    19	---
    20	
    21	A generated xgoja binary is a normal Go command-line program built from an `xgoja.yaml` file. It imports selected provider packages, registers their native Goja modules, embeds a normalized runtime specification, and creates commands that execute JavaScript against named runtime profiles.
    22	
    23	A runtime profile selects which provider modules are available to a command invocation. Commands such as `eval`, `run`, `repl`, and mounted JavaScript verbs create a fresh runtime for the selected profile, execute the requested JavaScript, and close the runtime afterwards. The `repl` command is the rich interactive terminal REPL, while `eval` is one-shot string evaluation.
    24	
    25	## Common commands
    26	
    27	- `modules` lists provider modules compiled into the binary.
    28	- `eval` or the configured evaluation command evaluates one JavaScript source string.
    29	- `run` executes a JavaScript file with the selected runtime profile and script-local module roots.
    30	- `repl` starts an interactive Bubble Tea REPL backed by the selected runtime profile.
    31	- `verbs` or the configured verbs command exposes JavaScript verbs from configured sources.
    32	- `help` lists bundled runtime help topics.
    33	
    34	## Runtime profile selection
    35	
    36	Commands that execute JavaScript accept or derive a runtime profile. The profile controls `require()` visibility. If a module is compiled into the binary but is not selected by the runtime profile, JavaScript code in that command cannot require it.

    22			return fmt.Errorf("start inference payload must be %T, got %T", &chatappv1.StartInferenceCommand{}, cmd.Payload)
    23		}
    24		pending := e.takePendingRequest(strings.TrimSpace(payload.GetRequestId()))
    25		prompt := strings.TrimSpace(pending.Prompt)
    26		if prompt == "" {
    27			prompt = strings.TrimSpace(payload.GetPrompt())
    28		}
    29		if prompt == "" {
    30			prompt = "Explain evtstream"
    31		}
    32		messageID := e.nextMessageID()
    33		userMessageID := messageID + "-user"
    34		if err := e.publish(ctx, cmd.SessionId, pub, EventUserMessageAccepted, &chatappv1.ChatUserMessageAccepted{MessageId: userMessageID, Role: "user", Text: prompt, Content: prompt, Status: "accepted"}); err != nil {
    35			return err
    36		}
    37		runCtx, cancel := context.WithCancel(publishContext(ctx))
    38		run := &activeRun{messageID: messageID, cancel: cancel, done: make(chan struct{})}
    39		if previous := e.swapRun(cmd.SessionId, run); previous != nil {
    40			previous.cancel()
    41			<-previous.done
    42		}
    43		go e.runPrompt(runCtx, cmd.SessionId, messageID, pending, prompt, pub, run.done)
    44		return nil
    45	}

    47	func (e *Engine) handleStopInference(_ context.Context, cmd sessionstream.Command, _ *sessionstream.Session, _ sessionstream.EventPublisher) error {
    48		if current := e.currentRun(cmd.SessionId); current != nil {
    49			current.cancel()
    50		}
    51		return nil
    52	}

    54	func (e *Engine) runPrompt(ctx context.Context, sid sessionstream.SessionId, messageID string, pending PromptRequest, prompt string, pub sessionstream.EventPublisher, done chan struct{}) {
    55		defer close(done)
    56		defer e.clearRun(sid, messageID)
    57		if pending.Runtime != nil && pending.Runtime.Engine != nil {
    58			e.runRuntimeInference(ctx, sid, messageID, prompt, pending, pub)
    59			return
    60		}
    61		e.runDemoInference(ctx, sid, messageID, prompt, pub)
    62	}

    64	func (e *Engine) runRuntimeInference(ctx context.Context, sid sessionstream.SessionId, messageID, prompt string, pending PromptRequest, pub sessionstream.EventPublisher) {
    65		runtime := pending.Runtime
    66		if runtime == nil || runtime.Engine == nil {
    67			e.runDemoInference(ctx, sid, messageID, prompt, pub)
    68			return
    69		}
    70		if err := e.publish(ctx, sid, pub, EventChatRunStarted, &chatappv1.ChatRunStarted{MessageId: messageID, Prompt: prompt, Correlation: runCorrelationInfo(sid, messageID)}); err != nil {
    71			return
    72		}
    73	
    74		baseSink := gepevents.EventSink(&runtimeEventSink{publishCtx: publishContext(ctx), sessionID: sid, messageID: messageID, prompt: prompt, pub: pub, engine: e})
    75		eventSink := baseSink
    76		if runtime.WrapSink != nil {
    77			wrapped, err := runtime.WrapSink(baseSink)
    78			if err != nil {
    79				e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
    80				return
    81			}
    82			eventSink = wrapped
    83		}
    84		sink, ok := baseSink.(*runtimeEventSink)
    85		if !ok {
    86			e.publishRunFailed(publishContext(ctx), sid, pub, messageID, "internal runtime sink type assertion failed")
    87			return
    88		}
    89		sess := gepsession.NewSessionWithID(string(sid))
    90		sess.Builder = &enginebuilder.Builder{
    91			Base:       runtime.Engine,
    92			EventSinks: []gepevents.EventSink{eventSink},
    93		}
    94	
    95		if pending.InitialTurn != nil {
    96			sess.Append(pending.InitialTurn.Clone())
    97		} else {
    98			// Load conversation history: the last persisted turn contains the full
    99			// conversation as an accumulator. AppendNewTurnFromUserPrompt will clone
   100			// it and add the new user block, giving the LLM the full context.
   101			if e.turnStore != nil {
   102				snapshot, err := e.turnStore.LoadLatestTurn(ctx, string(sid), "final")
   103				if err != nil {
   104					e.publishRunFailed(publishContext(ctx), sid, pub, messageID, fmt.Sprintf("load conversation history: %v", err))
   105					return
   106				}
   107				if snapshot != nil {
   108					turn, err := serde.FromYAML([]byte(snapshot.Payload))
   109					if err != nil {
   110						e.publishRunFailed(publishContext(ctx), sid, pub, messageID, fmt.Sprintf("decode conversation history: %v", err))
   111						return
   112					}
   113					if turn == nil {
   114						e.publishRunFailed(publishContext(ctx), sid, pub, messageID, "decode conversation history: empty turn")
   115						return
   116					}
   117					sess.Append(turn)
   118				}
   119			}
   120	
   121			_, err := sess.AppendNewTurnFromUserPrompt(prompt)
   122			if err != nil {
   123				e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
   124				return
   125			}
   126		}
   127		assistantBlockOffset := countAssistantBlocks(sess.Latest())
   128		handle, err := sess.StartInference(ctx)
   129		if err != nil {
   130			e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
   131			return
   132		}
   133		output, err := handle.Wait()
   134		if err != nil {
   135			if !sink.IsTerminal() {
   136				if errors.Is(err, context.Canceled) || ctx.Err() != nil {
   137					_ = sink.finishActiveTextSegment("stopped", "stopped", "")
   138					_ = e.publish(publishContext(ctx), sid, pub, EventChatRunStopped, &chatappv1.ChatRunStopped{MessageId: messageID, Status: "stopped", Correlation: runCorrelationInfo(sid, messageID)})
   139					return
   140				}
   141				_ = sink.finishActiveTextSegment("failed", "error", "")
   142				if isMaxIterationsError(err) {
   143					_ = e.publish(publishContext(ctx), sid, pub, EventChatTextSegmentFinished, &chatappv1.ChatTextSegmentFinished{MessageId: runtimeWarningMessageID(messageID), Role: "warning", Prompt: prompt, Text: maxIterationsWarningText(err), Content: maxIterationsWarningText(err), Status: "finished", Streaming: false, Final: true})
   144				}
   145				e.publishRunFailed(publishContext(ctx), sid, pub, messageID, err.Error())
   146			}
   147			return
   148		}
   149		if pending.OnFinalTurn != nil && output != nil {
   150			pending.OnFinalTurn(output.Clone())
   151		}
   152		if sink.IsTerminal() {
   153			return
   154		}
   155		if sink.HasActiveTextSegment() {
   156			_ = sink.finishActiveTextSegment("finished", "stop", "")
   157		} else if !sink.HasTextSegment() {
   158			_ = e.publishFallbackAssistantText(publishContext(ctx), sid, pub, messageID, prompt, output, assistantBlockOffset)
   159		}
   160		_ = e.publish(publishContext(ctx), sid, pub, EventChatRunFinished, &chatappv1.ChatRunFinished{MessageId: messageID, Status: "finished", Correlation: runCorrelationInfo(sid, messageID)})
   161	}

   166		}
   167		return context.WithoutCancel(ctx)

   179	func (e *Engine) publishFallbackAssistantText(ctx context.Context, sid sessionstream.SessionId, pub sessionstream.EventPublisher, messageID, prompt string, output *turns.Turn, assistantBlockOffset int) error {
   180		text := assistantTextFromTurnAfter(output, assistantBlockOffset)
   181		if strings.TrimSpace(text) == "" {
   182			return nil
   183		}
   184		const segment int32 = 1
   185		textMessageID := textSegmentMessageID(messageID, segment)
   186		corr := fallbackTextCorrelationInfo(sid, messageID, segment)
   187		if err := e.publish(ctx, sid, pub, EventChatTextSegmentStarted, &chatappv1.ChatTextSegmentStarted{MessageId: textMessageID, Role: "assistant", Prompt: prompt, Status: "streaming", Streaming: true, Correlation: corr}); err != nil {
   188			return err
   189		}
   190		return e.publish(ctx, sid, pub, EventChatTextSegmentFinished, &chatappv1.ChatTextSegmentFinished{MessageId: textMessageID, Role: "assistant", Prompt: prompt, Text: text, Content: text, Status: "finished", Streaming: false, Final: true, FinishReason: "stop", Correlation: corr})
   191	}

   193	func countAssistantBlocks(turn *turns.Turn) int {
   194		if turn == nil {
   195			return 0
   196		}
   197		count := 0
   198		for _, block := range turn.Blocks {
   199			if block.Role == turns.RoleAssistant {
   200				count++
   201			}
   202		}
   203		return count
   204	}

   206	func assistantTextFromTurnAfter(turn *turns.Turn, skip int) string {
   207		if turn == nil {
   208			return ""
   209		}
   210		parts := make([]string, 0, len(turn.Blocks))
   211		seen := 0
   212		for _, block := range turn.Blocks {
   213			if block.Role != turns.RoleAssistant {
   214				continue
   215			}
   216			if seen < skip {
   217				seen++
   218				continue
   219			}
   220			seen++
   221			text, _ := block.Payload[turns.PayloadKeyText].(string)
   222			if strings.TrimSpace(text) == "" {
   223				continue
   224			}
   225			parts = append(parts, text)
   226		}
   227		return strings.Join(parts, "")
   228	}
