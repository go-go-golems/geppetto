   100	// EventPartialCompletion is the event type for textual partial completion. We don't support partial tool completion.
   101	type EventPartialCompletion struct {
   102		EventImpl
   103		Delta string `json:"delta"`
   104		// This is the complete completion string so far (when using openai, this is currently also the toolcall json)
   105		Completion string `json:"completion"`
   106	
   107		// TODO(manuel, 2024-06-04) This might need partial tool completion if it is of interest,
   108		// this is less important than adding tool call information to the result above
   109	}
   110	
   111	func NewPartialCompletionEvent(metadata EventMetadata, delta string, completion string) *EventPartialCompletion {
   112		return &EventPartialCompletion{
   113			EventImpl: EventImpl{
   114				Type_:     EventTypePartialCompletion,
   115				Metadata_: metadata,
   116				payload:   nil,
   117			},
   118			Delta:      delta,
   119			Completion: completion,
   120		}
   121	}
   122	
   123	var _ Event = &EventPartialCompletion{}

   125	// EventThinkingPartial mirrors EventPartialCompletion but is dedicated to reasoning/summary text
   126	type EventThinkingPartial struct {
   127		EventImpl
   128		Delta      string `json:"delta"`
   129		Completion string `json:"completion"`
   130	}
   131	
   132	func NewThinkingPartialEvent(metadata EventMetadata, delta string, completion string) *EventThinkingPartial {
   133		return &EventThinkingPartial{
   134			EventImpl:  EventImpl{Type_: EventTypePartialThinking, Metadata_: metadata},
   135			Delta:      delta,
   136			Completion: completion,
   137		}
   138	}
   139	
   140	var _ Event = &EventThinkingPartial{}
