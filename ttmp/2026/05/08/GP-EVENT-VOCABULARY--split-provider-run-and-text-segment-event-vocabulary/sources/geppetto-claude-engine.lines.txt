   160		}
   161	
   162		completionMerger := NewContentBlockMerger(metadata)
   163	
   164		eventCount := 0
   165		for {
   166			select {
   167			case <-ctx.Done():
   168				log.Debug().Msg("Claude streaming cancelled by context")
   169				// Publish interrupt event with current partial text
   170				e.publishEvent(ctx, events.NewInterruptEvent(metadata, completionMerger.Text()))
   171				return nil, ctx.Err()
   172	
   173			case event, ok := <-eventCh:
   174				if !ok {
   175					log.Debug().Int("total_events", eventCount).Msg("Claude streaming channel closed, loop completed")
   176					goto streamingComplete
   177				}
   178	
   179				eventCount++
   180				e.observeProviderEvent(ctx, metadata, req.Model, event)
   181	
   182				events_, err := completionMerger.Add(event)
   183				if err != nil {
   184					log.Error().Err(err).Int("event_count", eventCount).Msg("Claude ContentBlockMerger.Add failed")
   185					e.publishEvent(ctx, events.NewErrorEvent(metadata, err))
   186					return nil, err
   187				}
   188				// Keep local metadata in sync even when Add returns no publishable
   189				// event. Claude message_delta/message_stop can carry stop_reason,
   190				// usage, and duration metadata while producing no transcript event.
   191				syncClaudeEventMetadata(&metadata, completionMerger.Metadata())
   192				// Publish intermediate events generated by the ContentBlockMerger.
   193				for _, event_ := range events_ {
   194					syncClaudeEventMetadata(&metadata, event_.Metadata())
   195					e.publishEvent(ctx, event_)
   196				}
   197			}
   198		}
   199	
   200	streamingComplete:
   201	
   202		response := completionMerger.Response()
   203		if response == nil {
   204			err := errors.New("no response")
   205			log.Error().Err(err).Msg("Claude ContentBlockMerger returned nil response")
   206			e.publishEvent(ctx, events.NewErrorEvent(metadata, err))
   207			return nil, err
   208		}
   209	
   210		log.Trace().Str("response_text", response.FullText()).Msg("Claude creating final message")
   211		// Update metadata with final response data without overwriting streamed
   212		// message_delta metadata with a zero-value message_start response.
   213		syncClaudeEventMetadata(&metadata, completionMerger.Metadata())
   214		if usage := claudeResponseUsageToEventUsage(response.Usage); usage != nil {
   215			metadata.Usage = usage
   216		}
   217		stopReason := ""
   218		if strings.TrimSpace(response.StopReason) != "" {
   219			sr := strings.TrimSpace(response.StopReason)
   220			metadata.StopReason = &sr
   221		}
   222		if metadata.StopReason != nil {
   223			stopReason = *metadata.StopReason
   224		}
   225		log.Trace().
   226			Interface("usage", metadata.Usage).
   227			Str("stop_reason", stopReason).
   228			Msg("Claude metadata finalized")
   229	
   230		// Create blocks from content blocks: text -> llm_text, tool_use -> tool_call
   231		hasToolCalls := false
   232		for _, c := range response.Content {
   233			switch v := c.(type) {
   234			case api.TextContent:
   235				if s := v.Text; s != "" {
   236					turns.AppendBlock(t, turns.NewAssistantTextBlock(s))
   237				}
   238			case api.ToolUseContent:
   239				hasToolCalls = true
   240				var args any
   241				_ = json.Unmarshal(v.Input, &args)
   242				turns.AppendBlock(t, turns.NewToolCallBlock(v.ID, v.Name, args))
   243			}
   244		}
   245	
   246		result := engine.BuildInferenceResultFromEventMetadata(metadata, "claude", hasToolCalls)
   247		if err := engine.PersistInferenceResult(t, result); err != nil {
   248			log.Warn().Err(err).Msg("Claude: failed to persist canonical inference_result")
   249		}
   250	
   251		// NOTE: Final event is already published by ContentBlockMerger during event processing
   252		log.Trace().Msg("Claude RunInference completed (streaming)")
   253		return t, nil
   254	}
   255	
   256	func syncClaudeEventMetadata(dst *events.EventMetadata, src events.EventMetadata) {
   257		if dst == nil {
   258			return
   259		}
   260		if src.StopReason != nil && strings.TrimSpace(*src.StopReason) != "" {
   261			sr := strings.TrimSpace(*src.StopReason)
   262			dst.StopReason = &sr
   263		}
   264		if src.Usage != nil {
   265			dst.Usage = src.Usage
   266		}
   267		if src.DurationMs != nil {
   268			dst.DurationMs = src.DurationMs
   269		}
   270		if len(src.Extra) > 0 {
   271			if dst.Extra == nil {
   272				dst.Extra = map[string]interface{}{}
   273			}
   274			for k, v := range src.Extra {
   275				dst.Extra[k] = v
   276			}
   277		}
   278	}
   279	
   280	func claudeResponseUsageToEventUsage(u api.Usage) *events.Usage {
   281		if u.InputTokens == 0 && u.OutputTokens == 0 && u.CacheCreationInputTokens == 0 && u.CacheReadInputTokens == 0 {
   282			return nil
   283		}
   284		return &events.Usage{
   285			InputTokens:              u.InputTokens,
   286			OutputTokens:             u.OutputTokens,
   287			CacheCreationInputTokens: u.CacheCreationInputTokens,
   288			CacheReadInputTokens:     u.CacheReadInputTokens,
   289		}
   290	}
